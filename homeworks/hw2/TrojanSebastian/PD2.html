library(scales)
library(forcats)
library(dplyr)
library(ggplot2)
library(tidyr)
nazwiska <- c("Joanna Tyrowicz","Ludwik Kotecki","Przemysaw Litwiniuk","Henryk Wnorowski", "Adam Glapiski", "Cezary Kochalski", "Iwona Duda", "Gabriela Masowska","Wiesaw Janczyk","Ireneusz Dbrowski")
Nastawienie_paz = c(5,4.6,3.7,2.7,2.4,2.4,2.3,2.1,2.0,1.9)
Zmiana = c(0,-0.2,0.8,-1.5,-1,-0.7,0,0,-0.4,-1.2)
dane <- data.frame(Nastawienie_czerw = Nastawienie_paz-Zmiana,
                   Nastawienie_paz = Nastawienie_paz,
           nazwiska = nazwiska)

wykres_poprawiony <- dane %>%
  mutate(nazwiska = fct_reorder(nazwiska, Nastawienie_paz, .desc = TRUE)) %>%
  pivot_longer(!nazwiska,names_to = "nastawienie", values_to = "wartosci") %>%
  filter(nastawienie != "Zmiana") %>%
  ggplot(aes(x=nazwiska, y=wartosci, group = nastawienie, color = nastawienie)) +
  geom_point(alpha = 1, size = 7)+ 
  scale_color_manual(
    values = c("#808080", "black"),
    labels = c("konsensus z czerwca '22", "konsensus z padziernika '22")) +
  theme(
    panel.background = element_rect(fill = 'white'),
    panel.grid.major.y = element_line(size = 1, color = "grey"),
    panel.grid.minor.y = element_line(size = 0.2, color = "grey"),
    panel.grid.major.x = element_line(size = 1, color = "black"),
    legend.title = element_blank(),
    legend.position = c(0.87,1.08),
    legend.background = element_blank(),
    axis.text.x = element_text(size = 15),
    axis.text.y = element_text(size = 15),
    axis.title = element_text(size = 20),
    plot.title = element_text(size = 25),
    plot.subtitle = element_text(size = 15),
    legend.text = element_text(size =12 )
  ) +
  scale_x_discrete(labels = wrap_format(10)) +
  labs(
    title = "Kosensus PAP: gobie i jastrzbie w RPP",
    subtitle = "Konsensus PAP dotyczcy nastawienia poszczeglnych czonkw RPP do zmian w polityce pieninej",
    y = "Nastawienie",
    x = NULL
  )+
  scale_y_continuous(
    limits = c(1, 5), labels  = c("gobie - 1","2","neutralne - 3","4","jastrzbie - 5")
    )

Uzasadnienie

Poprawniane przeze mnie bdy: Na poprawianym wykresie informacja o nastawieniu danej osoby jest przedstawiona przy pomocy a trzech narzdzi: wyokoci na jakiej znajude si punkt odpowiadajcy tej osobie, koloru tego punktu oraz liczby napisanej pod nazwiskiem tej osoby, podobnie na dwa sposoby przedstwaniona jest rnica pomidzy jej nastaweniem w poszczeglnych miesicach. Ponadto odczytanie danych z wykresu utrudnia duy rozmiar punktw, znajdujce si na nich zdjcia czonkw RPP oraz brak dokadniej skali na osi y.
